SELECT Tasks.*, Applications.entityID, Applications.applicationType, Applications.applicationName, Clients.clientID
	FROM Clients 
		INNER JOIN ((Tasks INNER JOIN Applications ON Tasks.appID = Applications.appID) 
		INNER JOIN Entities ON Applications.entityID = Entities.entityID) ON Clients.clientID =	Entities.clientID
			WHERE Clients.clientID = {{appsmith.store.clientID}}
			AND taskStatus <> "Complete"
;

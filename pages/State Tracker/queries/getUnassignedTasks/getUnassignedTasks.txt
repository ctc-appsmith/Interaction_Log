SELECT Tasks.*, Applications.entityID, Applications.applicationType, Applications.applicationName, Clients.clientID, taskTypes.taskName, taskTypes.taskTypeID, Tasks.assignedTeam
	FROM taskTypes
		INNER JOIN (Clients 
								INNER JOIN ((Tasks 
											     INNER JOIN Applications ON Tasks.appID = Applications.appID) 
															INNER JOIN Entities ON Applications.entityID = Entities.entityID) ON 
								             Clients.clientID =	Entities.clientID) ON taskTypes.taskTypeID = Tasks.taskTypeID
			WHERE Clients.clientID = {{appsmith.store.clientID}}
			AND (('{{RadioGroup3.selectedOptionValue}}' = "All") OR ( Tasks.assignedTeam= '{{RadioGroup3.selectedOptionValue}}'))
			AND (('{{RadioGroup4.selectedOptionValue}}' = "All") OR ( Applications.applicationType= '{{RadioGroup4.selectedOptionValue}}'))

;

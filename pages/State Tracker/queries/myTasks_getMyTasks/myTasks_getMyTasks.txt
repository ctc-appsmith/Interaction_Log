SELECT Tasks.*, Applications.entityID, Applications.applicationType, Applications.applicationName, Clients.clientID, taskTypes.taskName, taskTypes.taskTypeID
	FROM taskTypes
		INNER JOIN (Clients 
								INNER JOIN ((Tasks 
											     INNER JOIN Applications ON Tasks.appID = Applications.appID) 
															INNER JOIN Entities ON Applications.entityID = Entities.entityID) ON 
								             Clients.clientID =	Entities.clientID) ON taskTypes.taskTypeID = Tasks.taskTypeID
			WHERE Clients.clientID = {{appsmith.store.clientID}}
			AND taskStatus <> "Complete"
			AND (('{{RadioGroup1.value}}' = "All") OR ( Applications.applicationType= '{{RadioGroup1.value}}')
	)
;
